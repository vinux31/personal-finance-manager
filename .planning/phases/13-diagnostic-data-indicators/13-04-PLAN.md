```ts
import type { NetWorthAccount } from '@/db/netWorth'
import type { Investment } from '@/db/investments'
import { currentValue } from '@/db/investments'
import {
  FINANCIAL_TYPES,
  THRESHOLDS,
  type IndikatorColor,
  type IndikatorResult,
} from './kesehatanTypes'

// ============================================================
// Helper: aset finansial total (denominator untuk DIAG-07 Rasio Investasi)
//
// Konsisten dengan kesehatanTier1.ts totalAsetFinansial:
//   = SUM(accounts WHERE type IN FINANCIAL_TYPES) + SUM(currentValue(inv))
//   - EXCLUDE: properti, kendaraan (per spec §4 + STATE.md decisions)
// ============================================================
function totalAsetFinansial(
  accounts: NetWorthAccount[],
  investments: Investment[],
): number {
  const fromAccounts = accounts
    .filter(a => (FINANCIAL_TYPES as readonly string[]).includes(a.type))
    .reduce((s, a) => s + a.balance, 0)
  const fromInv = investments.reduce((s, inv) => s + currentValue(inv), 0)
  return fromAccounts + fromInv
}

// ============================================================
// #7 computeRasioInvestasi (DIAG-07)
// ============================================================
//
// Formula: (investments + deposito) ÷ aset finansial
// Threshold: ≥40% hijau, 20-39% kuning, <20% merah
// Properti & kendaraan EXCLUDE dari denominator (sudah handled via FINANCIAL_TYPES filter)
// ============================================================

export function computeRasioInvestasi(
  accounts: NetWorthAccount[],
  investments: Investment[],
): IndikatorResult {
  const asetFinansial = totalAsetFinansial(accounts, investments)

  if (asetFinansial === 0) {
    // Edge case: belum punya aset finansial sama sekali → red default (high incentive to start).
    return {
      kind: 'compute',
      value: 0,
      color: 'red',
      display: '— (belum ada aset)',
    }
  }

  const deposito = accounts
    .filter(a => a.type === 'deposito')
    .reduce((s, a) => s + a.balance, 0)
  const invValue = investments.reduce((s, inv) => s + currentValue(inv), 0)
  const numerator = invValue + deposito

  const pct = (numerator / asetFinansial) * 100

  const color: IndikatorColor =
    pct >= THRESHOLDS.rasioInvestasi.green
      ? 'green'
      : pct >= THRESHOLDS.rasioInvestasi.yellow
      ? 'yellow'
      : 'red'

  return {
    kind: 'compute',
    value: pct,
    color,
    display: `${pct.toFixed(0)}%`,
  }
}

// ============================================================
// #8 computeDiversifikasi (DIAG-08)
// ============================================================
//
// Formula: COUNT(DISTINCT asset_type normalized) + (1 if deposito balance > 0)
// Threshold: ≥3 hijau, 2 kuning, ≤1 merah
//
// asset_type normalization (per CONTEXT.md):
//   - lowercase + trim sebelum DISTINCT
//   - Partial mitigation Risk 5 (TEXT bebas).
//   - Examples:
//     "Saham BBCA" + "saham bbca" → 1 distinct type (post-normalize)
//     "Reksadana"  + "reksa dana"  → 2 distinct types (no semantic dedup di v1.2)
//   - Trade-off: partial protection vs over-engineering. Full normalize ke 5 standar
//     deferred ke v1.3 per spec out-of-scope.
//
// Deposito bonus:
//   - +1 jika ada minimal 1 account type='deposito' dengan balance > 0
//   - Deposito dianggap kelas aset terpisah dari investments
// ============================================================

export function computeDiversifikasi(
  investments: Investment[],
  accounts: NetWorthAccount[],
): IndikatorResult {
  // Normalize asset_type: lowercase + trim
  // Filter investments dengan currentValue>0 (skip positions yang sudah di-close
  // tapi masih punya row dengan quantity=0 — meskipun listInvestments sudah filter
  // .gt('quantity', 0), defensive di sini juga).
  const distinctTypes = new Set<string>(
    investments
      .filter(inv => currentValue(inv) > 0)
      .map(i => i.asset_type.toLowerCase().trim())
      .filter(t => t.length > 0),  // skip empty string post-trim
  )

  const distinctCount = distinctTypes.size

  const hasDeposito = accounts.some(a => a.type === 'deposito' && a.balance > 0)
  const score = distinctCount + (hasDeposito ? 1 : 0)

  const color: IndikatorColor =
    score >= THRESHOLDS.diversifikasi.green
      ? 'green'
      : score >= THRESHOLDS.diversifikasi.yellow
      ? 'yellow'
      : 'red'

  return {
    kind: 'compute',
    value: score,
    color,
    display: `${score} kelas aset`,
  }
}
```

**Edge case verification:**

**computeRasioInvestasi:**
- Aset finansial = 0 (no accounts likuid/deposito + no investments) → red "— (belum ada aset)". User belum mulai = high risk default.
- Aset finansial > 0 tapi deposito + investments = 0 (semua di akun likuid) → 0% red. User belum invest sama sekali.
- Numerator > denominator: tidak mungkin (numerator subset denominator), tapi defensive math `pct` bisa jadi 100 max.
- Properti/kendaraan tidak include karena FINANCIAL_TYPES filter exclude.

**computeDiversifikasi:**
- 0 investments + no deposito → score 0 → red.
- 0 investments + 1 deposito (balance > 0) → score 1 → red.
- 1 investment "Saham BBCA" + 1 deposito → score 2 → kuning.
- 2 investments ("BBCA" + "GOTO" both Saham) → distinct types after normalize = `{ 'saham bbca', 'goto' }` ATAU asset_type field di DB yang user input. Kalau user input persis "Saham" untuk both → 1 distinct type. Behavior tergantung user input. Document.
- 3 investments dengan asset_type "Reksadana", "Saham", "Emas" + 1 deposito → score 4 → hijau.
- Empty asset_type (string kosong setelah trim): skip via `.filter(t => t.length > 0)` — defensive.
