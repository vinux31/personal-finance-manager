Test pattern (DO NOT use pgTAP — use existing convention):
```sql
BEGIN;
SET LOCAL row_security = off;  -- belt-and-suspenders for direct seeds
\echo 'SECTION X: title'

DO $$
DECLARE
  v_uid UUID := '00000000-0000-0000-0000-000000000aaa';
BEGIN
  INSERT INTO auth.users (id, email, encrypted_password, created_at, updated_at, aud, role)
  VALUES (v_uid, 'test@example.local', '', NOW(), NOW(), 'authenticated', 'authenticated')
  ON CONFLICT (id) DO NOTHING;

  -- Switch RLS context
  PERFORM set_config('request.jwt.claim.sub', v_uid::TEXT, true);

  -- Re-enable RLS for the actual SELECT we want to test
  -- (SET LOCAL row_security = off applies to outer session;
  --  we use SET LOCAL ROLE authenticated to actually exercise RLS)

  -- Assert via RAISE NOTICE
  IF (SELECT count(*) FROM ...) = expected THEN
    RAISE NOTICE 'PASS: ...';
  ELSE
    RAISE NOTICE 'FAIL: ...';
  END IF;

  -- Assert exception is raised
  BEGIN
    PERFORM <call that should raise>;
    RAISE NOTICE 'FAIL: expected exception not raised';
  EXCEPTION WHEN OTHERS THEN
    IF SQLSTATE = '42501' THEN
      RAISE NOTICE 'PASS: 42501 raised — %', SQLERRM;
    ELSE
      RAISE NOTICE 'FAIL: unexpected SQLSTATE % — %', SQLSTATE, SQLERRM;
    END IF;
  END;
END $$;

ROLLBACK;
```

Key references for testing RLS in psql:
- `SET LOCAL row_security = off` → bypasses RLS at session level (used for SEEDING).
- `SET LOCAL ROLE authenticated` → enforces RLS as authenticated role (used for assertion queries).
- `PERFORM set_config('request.jwt.claim.sub', uid, true)` → sets `auth.uid()` for SECURITY DEFINER functions.

```sql
-- ============================================================
-- Phase 05 Wave 1 SQL Integration Test: Security Hardening (SEC-02 + SEC-03 + SEC-04)
--
-- Run: psql "$DATABASE_URL" -f supabase/tests/05-tighten-rls.sql
-- (for local dev: psql "postgresql://postgres:postgres@localhost:54322/postgres" -f ...)
--
-- Validates migration 0017_tighten_rls.sql (must be applied to target DB before running).
--
-- Convention (mirrors 04-mark-bill-paid.sql):
--   - BEGIN ... ROLLBACK wrapper → no state ever persists.
--   - SET LOCAL row_security = off for SEEDING (superuser bypass), then
--     SET LOCAL ROLE authenticated + set_config('request.jwt.claim.sub', uid)
--     for ASSERTIONS that need RLS to be enforced.
--   - Output via RAISE NOTICE 'PASS:' / 'FAIL:' — humans scan, CI greps for "FAIL:".
-- ============================================================

BEGIN;

-- Allow seeds to bypass RLS at the session level. RLS is re-enforced for assertions
-- by switching role + setting JWT claim.
SET LOCAL row_security = off;

-- ============================================================
-- SHARED SEEDS — synthetic admin + non-admin user
-- ============================================================

DO $$
DECLARE
  v_admin_uid    UUID := '00000000-0000-0000-0000-000000000a01';
  v_user_uid     UUID := '00000000-0000-0000-0000-000000000a02';
  v_other_uid    UUID := '00000000-0000-0000-0000-000000000a03';
  v_cat_id       BIGINT;
  v_count        INT;
  v_seeded_users BOOLEAN := false;
BEGIN
  -- Try to seed test users; abort all sections gracefully if env disallows
  BEGIN
    INSERT INTO auth.users (id, email, encrypted_password, created_at, updated_at, aud, role)
    VALUES
      (v_admin_uid, 'phase5-admin@example.local', '', NOW(), NOW(), 'authenticated', 'authenticated'),
      (v_user_uid,  'phase5-user@example.local',  '', NOW(), NOW(), 'authenticated', 'authenticated'),
      (v_other_uid, 'phase5-other@example.local', '', NOW(), NOW(), 'authenticated', 'authenticated')
    ON CONFLICT (id) DO NOTHING;
    v_seeded_users := true;
  EXCEPTION WHEN OTHERS THEN
    RAISE NOTICE 'SKIP ALL SECTIONS: cannot seed auth.users (%). Probably restricted environment.', SQLERRM;
    RETURN;
  END;

  -- Mark admin (insert/update profile)
  INSERT INTO profiles (id, is_admin, display_name)
  VALUES (v_admin_uid, true,  'Phase5 Admin')
  ON CONFLICT (id) DO UPDATE SET is_admin = true;

  INSERT INTO profiles (id, is_admin, display_name)
  VALUES (v_user_uid,  false, 'Phase5 User')
  ON CONFLICT (id) DO UPDATE SET is_admin = false;

  INSERT INTO profiles (id, is_admin, display_name)
  VALUES (v_other_uid, false, 'Phase5 Other')
  ON CONFLICT (id) DO UPDATE SET is_admin = false;

  -- Add allowed_emails for both users (so non-admin signup tests aren't blocked)
  INSERT INTO allowed_emails (email)
  VALUES ('phase5-user@example.local'), ('phase5-other@example.local')
  ON CONFLICT (email) DO NOTHING;

  -- Pick a category to use for transactions (any will do — already seeded by 0004)
  SELECT id INTO v_cat_id FROM categories ORDER BY id LIMIT 1;
  IF v_cat_id IS NULL THEN
    RAISE NOTICE 'SKIP ALL SECTIONS: no categories seeded';
    RETURN;
  END IF;

  -- Seed sample transactions for admin (so aggregate has data to expose if guard is broken)
  INSERT INTO transactions (date, type, category_id, amount, note, user_id)
  VALUES (CURRENT_DATE, 'expense', v_cat_id, 123456, 'phase5-admin-test', v_admin_uid);

  INSERT INTO transactions (date, type, category_id, amount, note, user_id)
  VALUES (CURRENT_DATE, 'expense', v_cat_id, 999, 'phase5-user-test', v_user_uid);

  -- ============================================================
  -- SECTION 1: SEC-02 / H-04 — profiles + allowed_emails RLS
  -- ============================================================

  RAISE NOTICE 'SECTION 1: profiles + allowed_emails RLS (SEC-02 / H-04)';

  -- Switch to non-admin user context: RLS now applies
  SET LOCAL ROLE authenticated;
  PERFORM set_config('request.jwt.claim.sub', v_user_uid::TEXT, true);

  -- Test 1.1 — non-admin sees own profile (and only own)
  SELECT count(*) INTO v_count FROM profiles;
  IF v_count = 1 THEN
    RAISE NOTICE 'PASS: non-admin sees exactly 1 profile row (own)';
  ELSE
    RAISE NOTICE 'FAIL: non-admin sees % profile rows (expected 1)', v_count;
  END IF;

  -- Test 1.2 — non-admin cannot see admin's profile row
  SELECT count(*) INTO v_count FROM profiles WHERE id = v_admin_uid;
  IF v_count = 0 THEN
    RAISE NOTICE 'PASS: non-admin cannot SELECT admin profile row';
  ELSE
    RAISE NOTICE 'FAIL: non-admin can SELECT admin profile row (RLS leaked)';
  END IF;

  -- Test 1.3 — non-admin sees empty allowed_emails
  SELECT count(*) INTO v_count FROM allowed_emails;
  IF v_count = 0 THEN
    RAISE NOTICE 'PASS: non-admin sees empty allowed_emails';
  ELSE
    RAISE NOTICE 'FAIL: non-admin sees % allowed_emails rows (expected 0)', v_count;
  END IF;

  -- Switch to admin: RLS allows everything
  PERFORM set_config('request.jwt.claim.sub', v_admin_uid::TEXT, true);

  -- Test 1.4 — admin sees all profile rows (>= 3 we seeded)
  SELECT count(*) INTO v_count FROM profiles;
  IF v_count >= 3 THEN
    RAISE NOTICE 'PASS: admin sees all profile rows (% >= 3)', v_count;
  ELSE
    RAISE NOTICE 'FAIL: admin sees % profile rows (expected >= 3)', v_count;
  END IF;

  -- Test 1.5 — admin sees allowed_emails
  SELECT count(*) INTO v_count FROM allowed_emails;
  IF v_count >= 1 THEN
    RAISE NOTICE 'PASS: admin sees allowed_emails (% >= 1)', v_count;
  ELSE
    RAISE NOTICE 'FAIL: admin sees % allowed_emails rows (expected >= 1)', v_count;
  END IF;

  -- ============================================================
  -- SECTION 2: SEC-03 / H-05 — allowlist bootstrap fail-closed
  -- ============================================================

  RAISE NOTICE 'SECTION 2: enforce_email_allowlist bootstrap fail-closed (SEC-03 / H-05)';

  -- Reset to superuser-equivalent for trigger testing (TRUNCATE bypasses our role)
  RESET ROLE;
  SET LOCAL row_security = off;

  -- Snapshot then wipe allowed_emails to simulate rogue DELETE/TRUNCATE
  CREATE TEMP TABLE _allowed_emails_snapshot AS SELECT * FROM allowed_emails;
  TRUNCATE allowed_emails;

  -- Test 2.1 — non-admin signup raises 'Allowlist kosong'
  BEGIN
    INSERT INTO auth.users (id, email, encrypted_password, created_at, updated_at, aud, role)
    VALUES ('00000000-0000-0000-0000-0000000000ee', 'attacker@evil.com', '', NOW(), NOW(), 'authenticated', 'authenticated');
    RAISE NOTICE 'FAIL: signup attacker@evil.com on empty allowlist did NOT raise';
  EXCEPTION WHEN OTHERS THEN
    IF SQLERRM LIKE '%Allowlist kosong%' THEN
      RAISE NOTICE 'PASS: empty-allowlist non-admin signup raised "%."', SQLERRM;
    ELSE
      RAISE NOTICE 'FAIL: unexpected error on empty-allowlist non-admin signup: %', SQLERRM;
    END IF;
  END;

  -- Test 2.2 — bootstrap admin signup succeeds even with empty allowlist
  -- (use unique uuid so we don't collide with any existing rinoadi28 user)
  BEGIN
    INSERT INTO auth.users (id, email, encrypted_password, created_at, updated_at, aud, role)
    VALUES ('00000000-0000-0000-0000-0000000000ad', 'rinoadi28@gmail.com', '', NOW(), NOW(), 'authenticated', 'authenticated')
    ON CONFLICT (email) DO NOTHING;
    RAISE NOTICE 'PASS: bootstrap admin signup allowed on empty allowlist';
  EXCEPTION WHEN OTHERS THEN
    -- ON CONFLICT DO NOTHING means if an existing rinoadi28 row exists, we silently skip.
    -- Any other error is a FAIL.
    RAISE NOTICE 'FAIL: bootstrap admin signup raised unexpectedly: %', SQLERRM;
  END;

  -- Restore allowed_emails for clean state in subsequent sections
  INSERT INTO allowed_emails SELECT * FROM _allowed_emails_snapshot;
  DROP TABLE _allowed_emails_snapshot;

  -- ============================================================
  -- SECTION 3: SEC-04 / H-06 — aggregate RPC IDOR guards
  -- ============================================================

  RAISE NOTICE 'SECTION 3: aggregate RPCs IDOR guard (SEC-04 / H-06)';

  -- Switch to non-admin context
  SET LOCAL ROLE authenticated;
  PERFORM set_config('request.jwt.claim.sub', v_user_uid::TEXT, true);

  -- Test 3.1 — non-admin calling aggregate_by_period(p_user_id := <admin uid>) raises 42501
  BEGIN
    PERFORM * FROM aggregate_by_period('month', NULL, NULL, v_admin_uid);
    RAISE NOTICE 'FAIL: aggregate_by_period IDOR (non-admin → admin uid) did NOT raise';
  EXCEPTION WHEN OTHERS THEN
    IF SQLSTATE = '42501' THEN
      RAISE NOTICE 'PASS: aggregate_by_period IDOR raised SQLSTATE 42501 — %', SQLERRM;
    ELSE
      RAISE NOTICE 'FAIL: aggregate_by_period IDOR raised wrong SQLSTATE % — %', SQLSTATE, SQLERRM;
    END IF;
  END;

  -- Test 3.2 — non-admin calling aggregate_by_category(p_user_id := <admin uid>) raises 42501
  BEGIN
    PERFORM * FROM aggregate_by_category('expense', NULL, NULL, v_admin_uid);
    RAISE NOTICE 'FAIL: aggregate_by_category IDOR (non-admin → admin uid) did NOT raise';
  EXCEPTION WHEN OTHERS THEN
    IF SQLSTATE = '42501' THEN
      RAISE NOTICE 'PASS: aggregate_by_category IDOR raised SQLSTATE 42501 — %', SQLERRM;
    ELSE
      RAISE NOTICE 'FAIL: aggregate_by_category IDOR raised wrong SQLSTATE % — %', SQLSTATE, SQLERRM;
    END IF;
  END;

  -- Test 3.3 — non-admin calling aggregate_by_period(p_user_id := NULL) returns own data (no false positive)
  SELECT count(*) INTO v_count FROM aggregate_by_period('month', NULL, NULL, NULL);
  -- v_count is rows in result; for non-admin user with 1 expense today, expect 1 row
  IF v_count >= 0 THEN
    RAISE NOTICE 'PASS: non-admin can call aggregate_by_period(NULL) for own data (% rows returned)', v_count;
  ELSE
    RAISE NOTICE 'FAIL: aggregate_by_period(NULL) failed for non-admin';
  END IF;

  -- Test 3.4 — non-admin calling with own uid explicitly (p_user_id = auth.uid()) succeeds
  SELECT count(*) INTO v_count FROM aggregate_by_period('month', NULL, NULL, v_user_uid);
  IF v_count >= 0 THEN
    RAISE NOTICE 'PASS: non-admin calling aggregate_by_period(p_user_id = own) succeeds';
  ELSE
    RAISE NOTICE 'FAIL: aggregate_by_period(p_user_id = own) for non-admin failed';
  END IF;

  -- ============================================================
  -- SECTION 4: T-05 — Admin View-As regression guard
  -- ============================================================

  RAISE NOTICE 'SECTION 4: Admin View-As regression (T-05)';

  -- Switch to admin context
  PERFORM set_config('request.jwt.claim.sub', v_admin_uid::TEXT, true);

  -- Test 4.1 — admin calling aggregate_by_period(p_user_id := <other user uid>) succeeds
  BEGIN
    PERFORM * FROM aggregate_by_period('month', NULL, NULL, v_user_uid);
    RAISE NOTICE 'PASS: admin can call aggregate_by_period with arbitrary p_user_id (View-As works)';
  EXCEPTION WHEN OTHERS THEN
    RAISE NOTICE 'FAIL: admin aggregate_by_period for impersonated uid raised: %', SQLERRM;
  END;

  -- Test 4.2 — admin SELECT on profiles returns all rows (View-As precondition)
  SELECT count(*) INTO v_count FROM profiles;
  IF v_count >= 3 THEN
    RAISE NOTICE 'PASS: admin SELECT profiles returns all rows for View-As (% >= 3)', v_count;
  ELSE
    RAISE NOTICE 'FAIL: admin SELECT profiles returned only % rows', v_count;
  END IF;

  -- Test 4.3 — admin SELECT on allowed_emails returns rows
  SELECT count(*) INTO v_count FROM allowed_emails;
  IF v_count >= 1 THEN
    RAISE NOTICE 'PASS: admin SELECT allowed_emails returns rows (% >= 1)', v_count;
  ELSE
    RAISE NOTICE 'FAIL: admin SELECT allowed_emails returned 0 rows';
  END IF;

  RESET ROLE;
END $$;

\echo '============================================================'
\echo 'Phase 5 RLS test complete. Rolling back all changes.'
\echo 'Review output above for any FAIL: lines.'
\echo 'Expected: 13 PASS notices total (Section 1: 5, Section 2: 2, Section 3: 4, Section 4: 3 — wait, Section 4 has 3 = total 14).'
\echo '============================================================'

ROLLBACK;
```
